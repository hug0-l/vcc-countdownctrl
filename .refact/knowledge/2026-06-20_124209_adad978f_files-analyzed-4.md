  ## Spec

  ### API 設計

  ```javascript
  // 3 行程式碼接入！
  const clipper = new ClipperSDK({
      server: 'ws://localhost:8765',  // default
      room: '1234',                    // default
      displayName: '計時器'            // required
  });

  // 監聽事件
  clipper.on('chat', (msg) => { console.log(msg.from, msg.text); });
  clipper.on('notice', (post) => { /* 更新公告 */ });
  clipper.on('connected', () => { /* 已連線 */ });
  clipper.on('disconnected', () => { /* 斷線 */ });

  // 發送操作
  clipper.sendChat('大家好！');
  clipper.createNotice('標題', '內容', '重要', ['tag1']);
  clipper.getState();  // Promise<state>
  ```
